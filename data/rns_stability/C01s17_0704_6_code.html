def interpol(x1, y1, z1, N):
    '''
    x1, y1: x-y grid coord of the simulation
    z1: the simulation data to interpolate
    N: number of points for plotting/interpolation
    '''
    x1i = np.linspace(x1.min(), x1.max(), N)
    y1i = np.linspace(y1.min(), y1.max(), N)
    z1i = scipy.interpolate.griddata((x1, y1), 
                                        z1, (x1i[None,:], 
                                        y1i[:,None]),
                                        method='linear')
    return x1i, y1i, z1i

    rho_xy_1d_level, x1d, y1d = cactus_hdf5_get_level_data(
        f'{root_dir}/output-000{num_dir_plot}/illinoisgrmhd-grmhd_primitives_allbutbi.xy.h5',

    xi, yi = np.meshgrid(np.arange(rho_xy_1d_level.shape[0]), np.arange(rho_xy_1d_level.shape[1]), 
                       sparse=False, indexing='xy')
    Qxx = np.sum(rho_xy_1d_level[xi, yi] * (3*x1d[xi]**2 - (x1d[xi]**2 + y1d[yi]**2)) * dx * dy)
